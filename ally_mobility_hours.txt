--- Version 0.18.1
-- Sourced from https://github.com/Ally-for-Hire/gmod-starfall-projects/blob/main/ally_mobility_hours.txt
-- Perf (0.18.1): cut the steady-state per-tick cost (~300us -> target sub-100us).
-- The per-tick drivetrain checks no longer call the ACF type predicates
-- (acfIsEngine/acfIsGearbox -- each a sandboxed ACF call that re-runs IsACFEntity
-- and was being made 2x for the gearbox and ~2x per engine EVERY tick); the
-- entity types can't change while the entities stay valid (a rewire/swap forces a
-- full refresh), so enginesReady/getEngineRPM/needsDrivetrainRefresh and the
-- fail-closed gate now check plain isValid and let updateEngineVariables remain
-- the one authoritative type gate. isValid is localized. Separately, the ~14
-- per-field-write pcalls in writeOutputs collapse to a single pcall around the
-- whole write batch (setField is now a plain write); same death-recovery
-- semantics (fail-closed next refresh), far less pcall overhead. No behavior change.
-- Drivetrain (0.18.0): engine discovery now recurses the gearbox graph from the
-- MainGearbox to ANY depth (not just one level), so engine -> gearbox -> ... -> CVT
-- layouts find their engines. The product of the intermediate gearboxes' current
-- total ratios (the engine -> CVT-input reduction) is captured as EngineUpstreamRatio
-- and folded into the gear-speed math, so the shift schedule is correct when the
-- engines are NOT wired straight to the CVT. EngineUpstreamRatio = 1 (unchanged
-- behavior) when they are direct.
-- Perf (0.17.1): per-tick wirelink writes now pass stable constant slot keys to
-- updateWirelinkField, so the backend stops rebuilding a tostring(wl)..field
-- change-cache key (string garbage) on every write. resetChanged() is called on
-- each drivetrain refresh so a relinked wirelink isn't suppressed by a stale
-- role-keyed value. No behavior change; writes were already value-suppressed.
-- Predictive counter-steer: the straight-line yaw corrector now projects the
-- hull's yaw rate forward by COUNTERSTEER_LOOKAHEAD seconds (using smoothed
-- yaw acceleration) and corrects on that predicted value, so the brake leads
-- the slew instead of chasing it. Set COUNTERSTEER_LOOKAHEAD = 0 to revert to
-- the old purely reactive behavior.
-- Death-hardening: every wirelink write is pcall-protected, because a wirelink
-- to a dead entity throws "Object no longer valid" on any access (so an
-- isValid(wl) probe cannot guard it -- the probe itself raises). The Pod
-- wirelink is isValid-gated in fail-closed, and stale engine/transfer writes
-- are swallowed instead of indexing into a NULL entity. The chip drops to
-- fail-closed on the next tick when any cached entity dies.

--[[
Tracked / wheeled ACF mobility controller.
Inputs:  MainGearbox, Base, Pod wirelink.
Outputs: LeftWeldLatch, RightWeldLatch.
Engines are auto-discovered by recursively walking MainGearbox:acfLinks() through
any depth of intermediate gearboxes / transfer cases (accumulating their ratios)
and driven directly via their own wirelinks, so no Engine wire input or Throttle
wire output is required.

Drive convention (applied in both neutral-steer and brake-steer branches):
  A / left  -> brake / disengage the LEFT side.
  D / right -> brake / disengage the RIGHT side.
If a vehicle is wired backwards, flip INVERT_BRAKE_STEER (brake-steer chassis)
or INVERT_COUNTERSTEER (yaw correction) to swap that branch's polarity.
]]--

require("allylib/simple_wire_helper.txt")
require("allylib/general_algorithms.txt")

-- Hot-path localization. The per-tick drivetrain validity checks below were
-- switched from the ACF type predicates (acfIsEngine/acfIsGearbox -- each a
-- sandboxed ACF call that re-runs IsACFEntity internally) to plain isValid:
-- an entity that passed the ACF type check at DISCOVERY is still that type for
-- as long as it stays valid, and a rewire/swap of the MainGearbox input already
-- forces a full refresh (inputClk dirty flag + the MainGearbox ~= PrevMainGearbox
-- guard). So per tick we only need "is it still alive?", not "is it still an
-- engine?". isValid is localized because it is now the most-called per-tick op.
local isValid = isValid

updateInputs("MainGearbox:ENTITY Base:ENTITY Pod:WIRELINK")
unnetworkInputs(true)
updateOutputs("LeftWeldLatch:NORMAL RightWeldLatch:NORMAL")

----------------------------------------------------------------------
-- Wire writers
----------------------------------------------------------------------
-- `slot` is a stable changed()-cache key threaded into updateWirelinkField.
-- Without it the backend rebuilds "__wl:"..tostring(wl)..":"..field on EVERY
-- write (two strings of garbage each tick); a constant literal slot avoids
-- that. The key must encode the wirelink ROLE (main/L/R/engine), because e.g.
-- LeftWL and RightWL both write "Left Brake" and would otherwise collide on a
-- single field-keyed slot. resetChanged() on every drivetrain refresh keeps a
-- relinked wirelink from being suppressed by a stale role-keyed value.
-- A wirelink whose entity has died throws "Object no longer valid" on ANY access
-- (including isValid(wl) -- the probe itself raises), so these writes still need
-- pcall protection. But the pcall is now hoisted to the CALLER (writeOutputs wraps
-- the whole tick's writes in one pcall; stopStaleEngineWires wraps its own), so a
-- mid-tick entity death aborts the rest of that batch and the chip drops to
-- fail-closed on the next refresh -- the same recovery the per-write pcall gave,
-- but ~14 pcalls/tick collapse to 1. setField itself is now a plain write.
local function setField(wl, field, value, slot)
    if not wl then return end
    updateWirelinkField(wl, field, value, slot)
end

local function setBothFields(wl, fieldA, fieldB, value, slotA, slotB)
    setField(wl, fieldA, value, slotA)
    setField(wl, fieldB, value, slotB)
end

local function setSideGear(wl, gear, slotInline, slotLeft, slotRight)
    if INLINE_TRANSFERS then
        setField(wl, "Gear", gear, slotInline)
    else
        setBothFields(wl, "Left Gear", "Right Gear", gear, slotLeft, slotRight)
    end
end

-- Drive every discovered engine via its own wirelink. The chip no longer
-- exposes a Throttle wire output; throttle and Active are routed directly.
--
-- Gate: throttle is only ever 0 / 100 / IDLE and active is 0 / 1, so the pair is
-- piecewise-constant -- it changes only when driver input transitions. With many
-- engines (14+) the 2*N updateWirelinkField calls per tick dominate the loop even
-- though changed() suppresses the actual wire write, because the CALL overhead is
-- paid regardless. So skip the whole loop while (throttle, active) is unchanged;
-- the engines already hold those values. resetDrivetrainState() clears the gate
-- (sets LastEngineThrottle = nil) so the first write after a refresh/relink
-- force-pushes to the rebuilt engine list (which also had resetChanged() run).
local LastEngineThrottle, LastEngineActive = nil, nil
local function writeEngineWires(throttle, active)
    local activeNum = active and 1 or 0
    if throttle == LastEngineThrottle and activeNum == LastEngineActive then return end
    LastEngineThrottle, LastEngineActive = throttle, activeNum
    for i = 1, EngineCount do
        local wl = EngineWLs[i]
        if wl then
            setField(wl, "Throttle", throttle,  EngineThrottleSlots[i])
            setField(wl, "Active",   activeNum, EngineActiveSlots[i])
        end
    end
end

local function snapshotEngineWLs()
    local snapshot = {}
    for i = 1, #EngineWLs do snapshot[i] = EngineWLs[i] end
    return snapshot
end

local function hasCurrentEngineWL(wl)
    for i = 1, EngineCount do
        if EngineWLs[i] == wl then return true end
    end
    return false
end

local function stopStaleEngineWires(previousWLs)
    for i = 1, #previousWLs do
        local wl = previousWLs[i]
        if wl and not hasCurrentEngineWL(wl) then
            -- setField no longer self-pcalls; a stale snapshot wirelink can point
            -- at a dead entity, so guard each zeroing write (refresh path, rare).
            pcall(setField, wl, "Throttle", 0)
            pcall(setField, wl, "Active",   0)
        end
    end
end

-- Every wirelink write for the tick. setField is now a plain write; this whole
-- function is run inside ONE pcall by writeOutputs, so a mid-tick entity death
-- aborts the rest of the batch and the chip fail-closes next refresh (same
-- recovery as the old per-write pcall, but one pcall instead of ~14).
local function writeWirelinkOutputs()
    writeEngineWires(OutputThrottle, EngineReady)

    if not MainWL then return end
    setField(MainWL, "Gear", OutputGear, "mGear")

    if CanNeutralSteer then
        if not LeftWL or not RightWL then return end
        setBothFields(LeftWL,  "Left Brake",  "Right Brake",  OutputLeftBrake,   "lLB", "lRB")
        setBothFields(RightWL, "Left Brake",  "Right Brake",  OutputRightBrake,  "rLB", "rRB")
        setBothFields(LeftWL,  "Left Clutch", "Right Clutch", OutputLeftClutch,  "lLC", "lRC")
        setBothFields(RightWL, "Left Clutch", "Right Clutch", OutputRightClutch, "rLC", "rRC")
        setSideGear(LeftWL,  OutputLeftGear,  "lG", "lLG", "lRG")
        setSideGear(RightWL, OutputRightGear, "rG", "rLG", "rRG")
    elseif DetectedDualClutch then
        setField(MainWL, "Left Brake",   OutputLeftBrake,   "mLB")
        setField(MainWL, "Right Brake",  OutputRightBrake,  "mRB")
        setField(MainWL, "Left Clutch",  OutputLeftClutch,  "mLC")
        setField(MainWL, "Right Clutch", OutputRightClutch, "mRC")
    else
        setField(MainWL, "Brake",  OutputLeftBrake,  "mBrake")
        setField(MainWL, "Clutch", OutputLeftClutch, "mClutch")
    end
end

local function writeOutputs()
    LeftWeldLatch  = OutputLeftBrake  > 0 and 1 or 0
    RightWeldLatch = OutputRightBrake > 0 and 1 or 0

    -- One pcall guards the whole write batch (see writeWirelinkOutputs note).
    pcall(writeWirelinkOutputs)

    if DEBUG_GEARS and OutputGear ~= PrevWrittenGear then
        print("AMH shift -> Gear", OutputGear, "slot", OutputGearSlot, "auto", IsAutogearing)
        PrevWrittenGear = OutputGear
    end
end

----------------------------------------------------------------------
-- Output mutators

local function setBothSides(brake, clutch)
    OutputLeftBrake,  OutputRightBrake  = brake, brake
    OutputLeftClutch, OutputRightClutch = clutch, clutch
end

local function addLeftSide(brake, clutch)
    OutputLeftBrake  = math.max(OutputLeftBrake,  brake)
    OutputLeftClutch = math.max(OutputLeftClutch, clutch)
end

local function addRightSide(brake, clutch)
    OutputRightBrake  = math.max(OutputRightBrake,  brake)
    OutputRightClutch = math.max(OutputRightClutch, clutch)
end

local function addTurnBrake(turnDir, brake, clutch)
    if turnDir > 0 then addRightSide(brake, clutch) else addLeftSide(brake, clutch) end
end

local function flipTransferGear(g)
    return g == FORWARD_TRANSFER_GEAR and REVERSE_TRANSFER_GEAR or FORWARD_TRANSFER_GEAR
end

----------------------------------------------------------------------
-- ACF helpers
-- Class-string checks first; ACF starfall methods throw on non-ACF ents.
----------------------------------------------------------------------
local function isInputEntity(ent)
    return isValid(ent) and ent ~= chip()
end

-- isValid(wirelink) raises "Object no longer valid" when the wrapped entity has
-- died, so a wirelink validity check has to be pcall-wrapped to fail closed.
local function isWirelinkAlive(wl)
    if not wl then return false end
    local ok, valid = pcall(isValid, wl)
    return ok and valid == true
end

local function isACFEntity(ent)
    if not isValid(ent) then return false end
    local class = ent:getClass()
    return class ~= nil and class:find("acf_", 1, true) == 1
end

local function isEngine(ent)
    return isACFEntity(ent) and ent:acfIsEngine()
end

local function isGearbox(ent)
    return isACFEntity(ent) and ent:acfIsGearbox()
end

local function isDualClutch(ent)
    return isGearbox(ent) and ent:acfIsDual()
end

local function isTransferCase(ent)
    if not isACFEntity(ent) then return false end
    local t = ent:acfType()
    return t == "Transfer Case" or t == "Transfer"
end

local function gearboxTotalRatio(ent, gear)
    if not isValid(ent) then return 1 end
    local r = (ent:acfGearRatio(gear or 1) or 1) * (ent:acfFinalRatio() or 1)
    if r == 0 then return 1 end
    return math.abs(r)
end

-- Current effective ratio of a gearbox (current gear * final drive). Used for the
-- intermediate gearboxes BETWEEN the engine and the CVT, which the chip does not
-- shift -- so read their live ratio (acfTotalRatio = This.GearRatio) instead of
-- assuming a gear. Falls back to 1 (no reduction) on an invalid/zero ratio.
local function gearboxCurrentRatio(ent)
    if not isValid(ent) then return 1 end
    local r = ent:acfTotalRatio()
    if not r or r == 0 then return 1 end
    return math.abs(r)
end

----------------------------------------------------------------------
-- Engine bookkeeping
----------------------------------------------------------------------
local function containsEntity(tbl, ent)
    for i = 1, #tbl do if tbl[i] == ent then return true end end
    return false
end

local function clearArray(t)
    for i = 1, #t do t[i] = nil end
end

local function addEngine(ent)
    if not isEngine(ent) or containsEntity(Engines, ent) then return end

    local wl = ent:getWirelink()
    if not wl then return end

    local powerMin = ent:acfPowerbandMin() or 0
    local powerMax = ent:acfPowerbandMax() or 0
    local weight = math.max(ent:acfMaxTorque() or 0, 1)
    EngineCount = EngineCount + 1
    Engines[EngineCount]       = ent
    EngineWeights[EngineCount] = weight
    EngineWLs[EngineCount]     = wl
    -- Precompute stable per-engine slot keys ONCE at discovery so the per-tick
    -- engine writes don't rebuild a tostring(wl)..field cache key every tick.
    EngineThrottleSlots[EngineCount] = "eT" .. EngineCount
    EngineActiveSlots[EngineCount]   = "eA" .. EngineCount
    EngineWeight = EngineWeight + weight
    PowerMin = PowerMin + powerMin * weight
    PowerMax = PowerMax + powerMax * weight

    if not ActiveEngine or powerMax > (ActiveEngine:acfPowerbandMax() or 0) then
        ActiveEngine = ent
    end

    return true
end

-- Recursively walk the gearbox graph from the MainGearbox toward the engines,
-- crossing ANY number of intermediate gearboxes / transfer cases. `ratioToMain`
-- is the cumulative ratio from the node we're exploring up to the MainGearbox
-- input -- the product of the current total ratios of the gearboxes BETWEEN this
-- node and the MainGearbox. When an engine is reached, that product is its
-- engine -> CVT-input reduction; the first one found is captured as
-- EngineUpstreamRatio and folded into the gear-speed math, so a drivetrain wired
-- engine -> gearbox -> CVT reads the right wheel speed (not just engine -> CVT).
-- Downstream L/R transfers are also walked but terminate at wheels (no engine),
-- so they never set EngineUpstreamRatio; their ratio is captured by findTransfers.
local function discoverEnginesFrom(node, ratioToMain, visited, depth)
    local links = node:acfLinks()
    for i = 1, #links do
        local link = links[i]
        if isEngine(link) then
            if addEngine(link) and EngineUpstreamRatio == nil then
                EngineUpstreamRatio = ratioToMain
            end
        elseif (isGearbox(link) or isTransferCase(link))
            and not visited[link] and depth < MAX_DRIVETRAIN_DEPTH then
            visited[link] = true
            -- Engine RPM passes through `link` before reaching `node`, so `link`'s
            -- current total ratio multiplies into the engine -> CVT reduction.
            discoverEnginesFrom(link, ratioToMain * gearboxCurrentRatio(link), visited, depth + 1)
        end
    end
end

local function discoverEngines()
    local visited = {}
    visited[MainGearbox] = true
    discoverEnginesFrom(MainGearbox, 1, visited, 0)
end

local function enginesReady()
    if EngineCount < 1 then return false end
    for i = 1, EngineCount do
        -- isValid, not isEngine: addEngine already proved these are ACF engines.
        -- A still-valid engine is still an engine; this runs every tick.
        if not isValid(Engines[i]) or EngineWLs[i] == nil then return false end
    end
    return true
end

-- RPM is read once per tick and used only for clutch-release timing and the idle
-- gate, both relative to the (weighted) powerband. Engines on a common gearbox
-- input shaft are mechanically locked to the same RPM, so a single representative
-- sample is accurate -- and at 14 engines, one acfRPM() instead of fourteen is the
-- single biggest per-tick saving. ActiveEngine is the highest-powerband engine
-- captured at discovery; fall back to the first still-valid engine if it died.
local function getEngineRPM()
    if isValid(ActiveEngine) then return ActiveEngine:acfRPM() or 0 end
    for i = 1, EngineCount do
        local e = Engines[i]
        if isValid(e) then return e:acfRPM() or 0 end
    end
    return 0
end

----------------------------------------------------------------------
-- Gear scheduling
----------------------------------------------------------------------
local function insertGearBySpeed(slots, speeds, count, gear, speed)
    local slot = count + 1
    while slot > 1 and speeds[slot - 1] > speed do
        slots[slot]  = slots[slot - 1]
        speeds[slot] = speeds[slot - 1]
        slot = slot - 1
    end
    slots[slot]  = gear
    speeds[slot] = speed
    return count + 1
end

local function updateShiftSpeeds(firstSlot, count)
    local minNextRPM = PowerMax > 0 and PowerMin / PowerMax or 0
    for i = 0, count - 1 do
        local slot = firstSlot + i
        local gearSpeed = GearSpeeds[slot] or 0
        local shift = gearSpeed * GEAR_SHIFT_MARGIN
        local nextSpeed = GearSpeeds[slot + 1]
        if i < count - 1 and nextSpeed then
            -- Prefer shifting where next gear lands in its powerband, but never
            -- past this gear's powerband-max speed; otherwise the chip lets the
            -- engine climb past redline waiting for a shift that requires a
            -- speed only attainable above the rev limiter.
            shift = math.min(math.max(shift, nextSpeed * minNextRPM), gearSpeed)
        end
        GearShiftSpeeds[slot] = shift
    end
end

local function selectGearSlotBySpeed(speed, minSlot, maxSlot)
    for slot = minSlot, maxSlot do
        if speed <= (GearShiftSpeeds[slot] or 0) then return slot end
    end
    return maxSlot
end

local function resolveOutputGear()
    if OutputGearSlot <= GearCount then
        return GearSlots[OutputGearSlot] or GearSlots[1] or 1
    end
    return ReverseGearSlots[OutputGearSlot - GearCount] or GearSlots[1] or 1
end

----------------------------------------------------------------------
-- Drivetrain refresh
----------------------------------------------------------------------
local function resetDrivetrainState()
    EngineStateDirty = false
    MainWL, LeftWL, RightWL     = nil, nil, nil
    LeftTransfer, RightTransfer = nil, nil
    ActiveEngine                = nil
    EngineCount, EngineWeight   = 0, 0
    PowerMin, PowerMax          = 0, 0
    TransferRatio               = 1
    EngineUpstreamRatio         = nil
    CanNeutralSteer             = false
    DetectedDualClutch          = false
    EngineReady                 = false
    GearCount, ReverseGears     = 0, 0
    PrevWrittenGear             = nil
    clearArray(Engines)
    clearArray(EngineWeights)
    clearArray(EngineWLs)
    clearArray(EngineThrottleSlots)
    clearArray(EngineActiveSlots)
    -- Drop the wire-write change cache so a relinked wirelink isn't suppressed
    -- by the PREVIOUS drivetrain's cached values (slots are keyed by role, not
    -- by wirelink identity, so they must be reset whenever the drivetrain is).
    resetChanged()
    -- Force the next writeEngineWires to push to the rebuilt engine list even if
    -- (throttle, active) matches the pre-refresh value (the gate above is keyed on
    -- the values, not on which engines exist).
    LastEngineThrottle, LastEngineActive = nil, nil
    clearArray(GearSpeeds)
    clearArray(GearShiftSpeeds)
    clearArray(GearSlots)
    clearArray(ReverseGearSlots)
    clearArray(ReverseGearSpeeds)
end

local function findTransfers(gearLinks)
    local transfers = 0
    for i = 1, #gearLinks do
        local link = gearLinks[i]
        if isTransferCase(link) then
            transfers = transfers + 1
            local y = MainGearbox:worldToLocal(link:getPos()).y
            if INVERTED_TRANSFERS then y = -y end
            if y < 0 then RightTransfer = link end
            if y > 0 then LeftTransfer  = link end
            if transfers == 1 then TransferRatio = gearboxTotalRatio(link, 1) end
        end
    end
    return transfers
end

local function buildGearSchedules()
    local totalGears = MainGearbox:acfNumGears()
    -- Top-of-powerband wheel surface speed in km/h for a given gear ratio.
    local commonVal = (PowerMax / 60) * WHEEL_DIAMETER * math.pi

    for i = 1, totalGears do
        local ratio = MainGearbox:acfGearRatio(i)
        if ratio ~= 0 then
            -- engine -> [intermediate gearboxes] -> CVT(gear i) -> [transfer] -> wheel
            local totalRatio   = (EngineUpstreamRatio or 1)
                * gearboxTotalRatio(MainGearbox, i) * TransferRatio
            local speedAtPower = math.abs(commonVal / totalRatio * SPEEDCONV)
            if ratio < 0 then
                ReverseGears = insertGearBySpeed(
                    ReverseGearSlots, ReverseGearSpeeds, ReverseGears, i, speedAtPower)
            else
                GearCount = insertGearBySpeed(
                    GearSlots, GearSpeeds, GearCount, i, speedAtPower)
            end
        end
    end

    for i = 1, ReverseGears do GearSpeeds[GearCount + i] = ReverseGearSpeeds[i] end
    updateShiftSpeeds(1, GearCount)
    updateShiftSpeeds(GearCount + 1, ReverseGears)
end

local function debugPrintGearSchedule()
    if not DEBUG_GEARS then return end
    local ratios = {}
    for i = 1, GearCount do ratios[i] = MainGearbox:acfGearRatio(GearSlots[i]) or 0 end
    print("engines", EngineCount, "powerband", PowerMin, PowerMax)
    print("main final", MainGearbox:acfFinalRatio() or 0, "transfer ratio", TransferRatio,
        "upstream ratio", EngineUpstreamRatio or 1)
    print("forward ratios", unpack(ratios))
    print("speed schedule", unpack(GearSpeeds))
    print("shift schedule", unpack(GearShiftSpeeds))
end

local function updateEngineVariables()
    local previousEngineWLs = snapshotEngineWLs()

    resetDrivetrainState()
    if not isGearbox(MainGearbox) then
        stopStaleEngineWires(previousEngineWLs)
        return
    end

    local gearLinks = MainGearbox:acfLinks()
    discoverEngines()
    stopStaleEngineWires(previousEngineWLs)
    if EngineCount < 1 or EngineWeight <= 0 then return end

    PowerMin   = PowerMin / EngineWeight
    PowerMax   = PowerMax / EngineWeight
    BiteRPM    = PowerMin * CLUTCH_START
    ReleaseRPM = PowerMin

    local transfers = findTransfers(gearLinks)
    buildGearSchedules()

    OutputGearSlot = clampValue(OutputGearSlot, 1, math.max(GearCount + ReverseGears, 1))
    OutputGear     = resolveOutputGear()
    -- Need at least two forward gears for autoshift; one-gear vehicles fall
    -- through to the manual branch and just pin slot 1.
    IsAutogearing  = GearCount > 1
    debugPrintGearSchedule()

    MainWL = MainGearbox:getWirelink()
    if isValid(LeftTransfer)  then LeftWL  = LeftTransfer:getWirelink()  end
    if isValid(RightTransfer) then RightWL = RightTransfer:getWirelink() end

    CanNeutralSteer    = transfers >= 2 and LeftWL ~= nil and RightWL ~= nil
    DetectedDualClutch = not CanNeutralSteer and isDualClutch(MainGearbox)
    EngineReady        = MainWL ~= nil and enginesReady() and PowerMax > 0
        and GearCount > 0 and (transfers < 2 or CanNeutralSteer)
end

----------------------------------------------------------------------
-- Per-tick output stages
----------------------------------------------------------------------
-- Clear all counter-steer PID + prediction state. Dropping CounterSteerWarm
-- forces the next engaged tick to seed accel at 0, so a stale CounterSteerPrevYaw
-- from before a steer/coast gap can't fabricate a huge yaw-acceleration spike.
local function resetCounterSteer()
    CounterSteerI, CounterSteerPrevError      = 0, 0
    CounterSteerPrevYaw, CounterSteerYawAccel = 0, 0
    CounterSteerWarm                          = false
end

local function applyFailClosed()
    OutputGearSlot = 1
    OutputGear     = GearSlots[1] or ReverseGearSlots[1] or 1
    OutputLeftGear, OutputRightGear = FORWARD_TRANSFER_GEAR, FORWARD_TRANSFER_GEAR
    OutputThrottle = 0
    setBothSides(MAX_BRAKE_POWER, CLUTCH_FULL)
    resetCounterSteer()
end

local function applyIdle()
    OutputThrottle = IDLE_THROTTLE
    setBothSides(MAX_BRAKE_POWER, CLUTCH_FULL)
end

local function computeBrakePower(speed)
    if not SCALE_BRAKES or speed <= 1 then return MAX_BRAKE_POWER end
    local topSpeed = BRAKE_SPEED_GATE
    if IsAutogearing and (GearSpeeds[GearCount] or 0) > 0 then
        topSpeed = GearSpeeds[GearCount] / 2
    end
    return MAX_BRAKE_POWER * math.min(topSpeed / speed, 1)
end

-- Soft clutch release: 1 (fully disengaged) until RPM crosses BiteRPM,
-- 0 (fully engaged) by the time RPM reaches ReleaseRPM.
local function clutchReleaseFromRPM(rpm)
    local range = ReleaseRPM - BiteRPM
    if range <= 0 then return 0 end
    return 1 - clampValue((rpm - BiteRPM) / range, 0, 1)
end

local function chooseGearSlot(driveDir, speed, neutralSteer)
    if driveDir == 0 then
        driveDir = OutputGearSlot > GearCount and -1 or 1
    end

    if not IsAutogearing then
        if neutralSteer then
            OutputGearSlot = 1
        elseif driveDir < 0 and ReverseGears > 0 then
            OutputGearSlot = GearCount + 1
        else
            OutputGearSlot = 1
        end
    else
        local minSlot, maxSlot = 1, GearCount
        if driveDir < 0 and not neutralSteer and ReverseGears > 0 then
            minSlot = GearCount + 1
            maxSlot = GearCount + ReverseGears
        end
        OutputGearSlot = selectGearSlotBySpeed(speed, minSlot, maxSlot)
    end

    OutputGear = resolveOutputGear()
    return driveDir
end

local function applyNeutralSteer(driveDir, steerDir, brakePower, hasDrive, hasSteer)
    if not hasDrive and hasSteer then
        -- Pure pivot: opposite transfer gears spin tracks in opposite directions.
        if steerDir < 0 then
            OutputLeftGear, OutputRightGear = REVERSE_TRANSFER_GEAR, FORWARD_TRANSFER_GEAR
        else
            OutputLeftGear, OutputRightGear = FORWARD_TRANSFER_GEAR, REVERSE_TRANSFER_GEAR
        end
        return
    end

    if driveDir < 0 then
        OutputLeftGear, OutputRightGear = REVERSE_TRANSFER_GEAR, REVERSE_TRANSFER_GEAR
    end

    -- A -> brake left, D -> brake right.
    if steerDir < 0 then
        OutputLeftBrake, OutputRightBrake = brakePower, 0
        OutputLeftClutch = CLUTCH_FULL
    elseif steerDir > 0 then
        OutputLeftBrake, OutputRightBrake = 0, brakePower
        OutputRightClutch = CLUTCH_FULL
    end
end

local function applyBrakeSteer(driveDir, steerDir, brakePower, hasDrive, hasSteer, speed, steerOnlyDrive)
    if CanNeutralSteer and driveDir < 0 and ReverseGears == 0 then
        OutputLeftGear, OutputRightGear = REVERSE_TRANSFER_GEAR, REVERSE_TRANSFER_GEAR
    end

    local canSteerThisTick = hasSteer
        and (hasDrive or speed > 1 or steerOnlyDrive)
        and (CanNeutralSteer or DetectedDualClutch)
    if not canSteerThisTick then return end

    local turnDir = steerDir
    if driveDir < 0 and INVERT_REVERSE_STEER then turnDir = -turnDir end
    if INVERT_BRAKE_STEER then turnDir = -turnDir end

    addTurnBrake(turnDir, brakePower, CLUTCH_FULL)
end

local function applyCounterSteer(driveDir, brakePower)
    local yawRate = Base:getAngleVelocity().z

    -- Predict the yaw rate COUNTERSTEER_LOOKAHEAD seconds ahead and run the
    -- whole corrector (magnitude AND side) on that instead of the raw current
    -- rate. Estimate yaw acceleration from the last sample, EMA-smooth it to
    -- tame 10 Hz tick noise, then extrapolate. On the first engaged tick we
    -- have no prior sample, so accel is forced to 0 -- otherwise a stale
    -- CounterSteerPrevYaw from before a steer/coast gap would inject a bogus
    -- spike. LOOKAHEAD = 0 collapses yawPredicted back to yawRate exactly.
    local rawAccel = CounterSteerWarm and (yawRate - CounterSteerPrevYaw) / THINK_DT or 0
    CounterSteerWarm     = true
    CounterSteerYawAccel = CounterSteerYawAccel
        + (rawAccel - CounterSteerYawAccel) * COUNTERSTEER_ACCEL_SMOOTH
    CounterSteerPrevYaw  = yawRate

    local yawPredicted = yawRate + CounterSteerYawAccel * COUNTERSTEER_LOOKAHEAD
    local yawAbs       = math.abs(yawPredicted)

    if yawAbs <= YAW_DEADZONE then
        CounterSteerI, CounterSteerPrevError = 0, 0
        return
    end

    CounterSteerI = CounterSteerI + yawAbs * THINK_DT
    local pidNorm = clampValue(
        (yawAbs / MAX_YAW_RATE) * COUNTERSTEER_STRENGTH
        + (CounterSteerI / MAX_YAW_RATE) * COUNTERSTEER_KI
        + ((yawAbs - CounterSteerPrevError) / THINK_DT / MAX_YAW_RATE) * COUNTERSTEER_KD,
        0, 1)
    CounterSteerPrevError = yawAbs

    local correctionBrake = math.min(
        math.max(COUNTERSTEER_MIN_BRAKE, pidNorm * brakePower),
        brakePower)

    local reverseBiteActive = false
    if CanNeutralSteer
        and ACTUAL_COUNTERSTEER_ENGAGEMENT > 0
        and ACTUAL_COUNTERSTEER_THRESHOLD < 1 then
        local biteRange  = 1 - ACTUAL_COUNTERSTEER_THRESHOLD
        local biteScale  = clampValue(ACTUAL_COUNTERSTEER_ENGAGEMENT, 0, 1)
        local biteAmount = clampValue((pidNorm - ACTUAL_COUNTERSTEER_THRESHOLD) / biteRange, 0, 1) * biteScale
        reverseBiteActive = biteAmount > 0
    end

    local yawWhileDriving = yawPredicted * driveDir
    if INVERT_COUNTERSTEER then yawWhileDriving = -yawWhileDriving end

    -- Match brake-steer convention: right correction uses right brake.
    if yawWhileDriving < 0 then
        addRightSide(correctionBrake, CLUTCH_FULL)
        if reverseBiteActive then OutputRightGear = flipTransferGear(OutputRightGear) end
    else
        addLeftSide(correctionBrake, CLUTCH_FULL)
        if reverseBiteActive then OutputLeftGear = flipTransferGear(OutputLeftGear) end
    end
end

----------------------------------------------------------------------
-- Refresh gating
----------------------------------------------------------------------
local function needsDrivetrainRefresh()
    if EngineStateDirty
        or MainGearbox ~= PrevMainGearbox
        or not isValid(MainGearbox)   -- isGearbox re-check happens in updateEngineVariables
        or not enginesReady()
        or not EngineReady
        or MainWL == nil then
        return true
    end
    if CanNeutralSteer then
        return not isValid(LeftTransfer) or not isValid(RightTransfer)
            or LeftWL == nil or RightWL == nil
    end
    return false
end

----------------------------------------------------------------------
-- Main loop
----------------------------------------------------------------------
-- Diagnostic counters (only touched when DEBUG_PERF). PerfRefreshes counts how
-- many fires in the current 1s window actually ran updateEngineVariables.
local PerfAccum, PerfMax, PerfCount, PerfRefreshes, PerfNextPrint = 0, 0, 0, 0, 0

local function mainLogic()
    -- 1) Refresh cached ACF/wirelink data only when inputs or cache is stale.
    if needsDrivetrainRefresh() then
        updateEngineVariables()
        PrevMainGearbox = MainGearbox
        if DEBUG_PERF then PerfRefreshes = PerfRefreshes + 1 end
    end

    -- 2) Fail closed: no throttle, fallback gear, full brake/clutch.
    if not isValid(MainGearbox) or not isInputEntity(Base) or not isWirelinkAlive(Pod) or not EngineReady then
        applyFailClosed()
        writeOutputs()
        return
    end

    -- 3) Read driver intent and vehicle state.
    local vel      = Base:getVelocity()
    local speed    = math.sqrt(vel.x * vel.x + vel.y * vel.y) * SPEEDCONV
    local rpm      = getEngineRPM()
    local driveDir = (Pod["W"] or 0) - (Pod["S"] or 0)
    local steerDir = (Pod["D"] or 0) - (Pod["A"] or 0)

    local neutralSteer   = CanNeutralSteer and not HAS_WHEELS and not DISABLE_NEUTRAL_STEER
    local brakeSteer     = not HAS_WHEELS and not neutralSteer
    local hasDrive       = driveDir ~= 0
    local hasSteer       = steerDir ~= 0
    -- Brake-steer chassis: pure steering also drives forward.
    local steerOnlyDrive = brakeSteer and not hasDrive and hasSteer
    if steerOnlyDrive then driveDir = 1 end
    local hasInput = hasDrive or (neutralSteer and hasSteer) or steerOnlyDrive

    -- 4) Reset per-tick side gears (steering branches override).
    OutputLeftGear, OutputRightGear = FORWARD_TRANSFER_GEAR, FORWARD_TRANSFER_GEAR

    -- 5) Idle-while-parked: warm engine without letting it move.
    if not hasInput and speed < 1 and rpm < BiteRPM then
        applyIdle()
        writeOutputs()
        return
    end

    -- 6) Pick brake power, then throttle + base clutch state.
    -- No WASD input always brakes; hasInput overrides below to release.
    local brakePower = computeBrakePower(speed)
    setBothSides(brakePower, CLUTCH_FULL)

    if hasInput then
        OutputThrottle = 100
        setBothSides(0, clutchReleaseFromRPM(rpm))
    else
        OutputThrottle = IDLE_THROTTLE
    end

    -- 7) Choose gear slot and resolve to a real ACF gear number.
    driveDir = chooseGearSlot(driveDir, speed, neutralSteer)

    -- 8) Steering.
    if neutralSteer then
        applyNeutralSteer(driveDir, steerDir, brakePower, hasDrive, hasSteer)
    elseif brakeSteer then
        applyBrakeSteer(driveDir, steerDir, brakePower, hasDrive, hasSteer, speed, steerOnlyDrive)
    end

    -- 9) Straight-line yaw correction (tracked vehicles, driving, no steer input).
    if not HAS_WHEELS and hasDrive and not hasSteer then
        applyCounterSteer(driveDir, brakePower)
    else
        resetCounterSteer()
    end

    -- 10) Single write point at the end keeps wire side effects predictable.
    writeOutputs()
end

-- High-precision monotonic clock for the diagnostic. Prefer systime; fall back
-- to realtime if this SF build doesn't expose it (both return seconds).
local _systime = timer.systime or timer.realtime

timer.create("AMH_mainLogicLoop", THINK_DT, 0, function()
    if not DEBUG_PERF then return mainLogic() end

    local t0 = _systime()
    mainLogic()
    local dt = _systime() - t0

    PerfAccum = PerfAccum + dt
    if dt > PerfMax then PerfMax = dt end
    PerfCount = PerfCount + 1

    local now = _systime()
    if PerfNextPrint == 0 then PerfNextPrint = now + 1 end
    if now >= PerfNextPrint then
        print(string.format(
            "AMH perf: avg %.0f us, max %.0f us, %d fires/s, %d refreshes/s, engines %d, ready %s",
            PerfAccum / PerfCount * 1e6, PerfMax * 1e6, PerfCount, PerfRefreshes,
            EngineCount, tostring(EngineReady)))
        PerfAccum, PerfMax, PerfCount, PerfRefreshes = 0, 0, 0, 0
        PerfNextPrint = now + 1
    end
end)
