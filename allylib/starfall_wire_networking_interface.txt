--[[
Simple Starfall wire + networking backend.

Public API
  Ports     updateInputs(str), updateOutputs(str)
            Inputs.<Name>, Outputs.<Name>, InputValues.<Name>
  NetVars   addNetworkedVariable, updateNetworkedVariable, updateNetworkedVariables
            getNetworkedVariableValue, NetVars.<Name>, NetVarValues.<Name>
  Triggers  inputClk(name, fn), netVarClk(name, fn)
  Change    changed(value, keep, slot, eps), resetChanged(slot)
            Omitted slots auto-key call sites, with helper direct-value fallback.
  Wire ops  safeWireCreate, safeWireDelete, updateWirelinkField
  Routing   enableConnectSend(bool), networkTargetSet(target)
  State     addNetworkedGunState, addNetworkedPropArmorState, addStateSend and
            their update* counterparts; read back via State.<Prefix>.<Field>

Design choices for stability:
- One small net message format. Budget preflight and retry/requeue, no streaming.
- One server flush timer that batches dirty netvars into a single net message,
  capped at MAX_RECORDS_PER_FLUSH per tick.
- Client announces readiness once; server sends a snapshot to that player and
  adds them to the ready set when connect-send is enabled.
- Default values come from the declared port type; no name heuristics.
]]--

-- Private state: bookkeeping the public API hides; not read outside this file.
local InputValidity = {}
local InputTriggers = {}
local InputTypes = {}
local NetVarTriggers = {}
local StateConfigs = {}
local ConnectSendEnabled = false
local ConnectSendTargets = {}
local ConnectSendAnnounced = false
local NetworkTargets = nil

local NET_MESSAGE = "SFWN_" .. chip():entIndex()
local FLUSH_INTERVAL = 0.05
local MAX_RECORDS_PER_FLUSH = 32
local SNAPSHOT_RETRY_COUNT = 3
local STATE_POLL_INTERVAL = 0.25
local CONNECT_RETRY_INTERVAL = 0.5
local CurTime = timer.curtime
local NET_BASE_BITS = 8 * 8
local STRING_TERM_BITS = 8
local VECTOR_BITS = 3 * 32
local ANGLE_BITS = 3 * 32
local ENTITY_BITS = 16 + 32

local TYPE_NIL = 0
local TYPE_BOOL = 1
local TYPE_NUMBER = 2
local TYPE_STRING = 3
local TYPE_VECTOR = 4
local TYPE_ANGLE = 5
local TYPE_ENTITY = 6
local TYPE_ARRAY = 7
local TYPE_MAP = 8

local ENTITY_TYPES = {
    Entity = true, Player = true, Vehicle = true,
    Weapon = true, NPC = true, Hologram = true
}

local PORT_TYPE_MAP = {
    normal = "NORMAL", number = "NORMAL", float = "NORMAL", n = "NORMAL",
    entity = "ENTITY", e = "ENTITY",
    vector = "VECTOR", v = "VECTOR",
    angle = "ANGLE", a = "ANGLE",
    array = "ARRAY", table = "ARRAY", t = "ARRAY",
    string = "STRING", s = "STRING",
    wirelink = "WIRELINK", wl = "WIRELINK"
}

local function normalizePortType(t)
    if type(t) ~= "string" or t == "" then return "NORMAL" end
    return PORT_TYPE_MAP[string.lower(t)] or string.upper(t)
end

local function defaultForPortType(portType)
    if portType == "ENTITY" then return chip() end
    if portType == "VECTOR" then return Vector() end
    if portType == "ANGLE" then return Angle() end
    if portType == "ARRAY" then return {} end
    if portType == "STRING" then return "" end
    if portType == "WIRELINK" then return nil end
    return 0
end

local function isVector(v)
    return type(v) == "Vector"
end

local function isAngle(v)
    return type(v) == "Angle"
end

local function isEntityValue(v)
    if not ENTITY_TYPES[type(v)] then return false end
    return v.isValid == nil or v:isValid()
end

local function isDenseArray(t)
    if type(t) ~= "table" then return false, 0 end
    local n = 0
    for k in pairs(t) do
        if type(k) ~= "number" or k < 1 or k % 1 ~= 0 then return false, 0 end
        if k > n then n = k end
    end
    for i = 1, n do
        if t[i] == nil then return false, 0 end
    end
    return true, n
end

local function clone(v)
    if isVector(v) then return Vector(v.x, v.y, v.z) end
    if isAngle(v) then return Angle(v.p, v.y, v.r) end
    if type(v) ~= "table" then return v end
    local out = {}
    for k, value in pairs(v) do out[k] = clone(value) end
    return out
end

local function same(a, b)
    if a == b then return true end
    if isVector(a) and isVector(b) then return a.x == b.x and a.y == b.y and a.z == b.z end
    if isAngle(a) and isAngle(b) then return a.p == b.p and a.y == b.y and a.r == b.r end
    if type(a) ~= "table" or type(b) ~= "table" then return false end
    for k, va in pairs(a) do
        if not same(va, b[k]) then return false end
    end
    for k in pairs(b) do
        if a[k] == nil then return false end
    end
    return true
end

local function inferTypeCode(v)
    if v == nil then return TYPE_NIL end
    local t = type(v)
    if t == "boolean" then return TYPE_BOOL end
    if t == "number" then return TYPE_NUMBER end
    if t == "string" then return TYPE_STRING end
    if isVector(v) then return TYPE_VECTOR end
    if isAngle(v) then return TYPE_ANGLE end
    if isEntityValue(v) then return TYPE_ENTITY end
    if t == "table" then
        if isDenseArray(v) then return TYPE_ARRAY end
        return TYPE_MAP
    end
    return TYPE_NIL
end

local function estimateStringBits(v)
    return #tostring(v or "") * 8 + STRING_TERM_BITS
end

local estimateValueBits
estimateValueBits = function(v, depth)
    depth = (depth or 0) + 1
    if depth > 8 then return 4 end

    local code = inferTypeCode(v)
    local bits = 4
    if code == TYPE_NIL then return bits end
    if code == TYPE_BOOL then return bits + 1 end
    if code == TYPE_NUMBER then return bits + 64 end
    if code == TYPE_STRING then return bits + estimateStringBits(v) end
    if code == TYPE_VECTOR then return bits + VECTOR_BITS end
    if code == TYPE_ANGLE then return bits + ANGLE_BITS end
    if code == TYPE_ENTITY then return bits + ENTITY_BITS end

    if code == TYPE_ARRAY then
        local n = math.min(#v, 1024)
        bits = bits + 16
        for i = 1, n do bits = bits + estimateValueBits(v[i], depth) end
        return bits
    end

    if code == TYPE_MAP then
        local count = 0
        for _ in pairs(v) do count = count + 1 end
        count = math.min(count, 1024)

        bits = bits + 16
        local written = 0
        for k, value in pairs(v) do
            written = written + 1
            if written > count then break end
            bits = bits + estimateValueBits(k, depth) + estimateValueBits(value, depth)
        end
        return bits
    end

    return bits
end

-- changed() / resetChanged()
local ChangedSlots = {}
local ChangedCallFrame = nil
local ChangedCallCounts = {}

local function isDirectSlotCandidate(value)
    local t = type(value)
    return t == "table" or isVector(value) or isAngle(value) or ENTITY_TYPES[t] == true
end

local function changedSlotFromTable(values, prefix, value)
    if not isDirectSlotCandidate(value) or type(values) ~= "table" then return nil end
    local match = nil
    for name, stored in pairs(values) do
        if stored == value then
            if match ~= nil then return nil end
            match = name
        end
    end
    if match == nil then return nil end
    return prefix .. tostring(match)
end

local function changedSlotFromHelperValue(value)
    local slot = changedSlotFromTable(InputValues, "__input:", value)
    if slot then return slot end
    slot = changedSlotFromTable(NetVarValues, "__netvar:", value)
    if slot then return slot end
    return changedSlotFromTable(LastOutputs, "__output:", value)
end

local function changedSlotFromCallsite()
    if type(debug) ~= "table" or type(debug.getinfo) ~= "function" then return nil end
    local ok, info = pcall(debug.getinfo, 4, "Sl")
    if not ok or type(info) ~= "table" then return nil end
    local src = info.short_src or info.source or "?"
    local key = tostring(src) .. ":" .. tostring(info.currentline or 0)
    local now = CurTime()
    if ChangedCallFrame ~= now then
        ChangedCallFrame = now
        ChangedCallCounts = {}
    end
    local count = (ChangedCallCounts[key] or 0) + 1
    ChangedCallCounts[key] = count
    return "__call:" .. key .. ":" .. count
end

local function defaultChangedSlot(value)
    return changedSlotFromCallsite()
        or changedSlotFromHelperValue(value)
        or ("__default:" .. tostring(type(value)))
end

function changed(value, keep, slot, epsilon)
    slot = slot or defaultChangedSlot(value)
    local prev = ChangedSlots[slot]
    local didChange
    if epsilon and type(value) == "number" and type(prev) == "number" then
        didChange = math.abs(value - prev) > epsilon
    else
        didChange = not same(value, prev)
    end
    if keep ~= false then ChangedSlots[slot] = clone(value) end
    return didChange
end

function resetChanged(slot)
    if slot ~= nil then
        ChangedSlots[slot] = nil
    else
        ChangedSlots = {}
    end
end

-- Port object prototypes
local inputProto = {}
inputProto.__index = inputProto

function inputProto:get()
    local v = InputValues[self.name]
    if v ~= nil then return v end
    if SERVER and wire.ports then return wire.ports[self.name] end
    return nil
end

function inputProto:isValid()
    if InputValidity[self.name] == true then return true end
    if not SERVER then return false end
    local ok, w = pcall(wire.self)
    if not ok then return false end
    return w:isWired(self.name) == true
end

function inputProto:addTrigger(fn)
    if type(fn) ~= "function" then return nil end
    local list = InputTriggers[self.name] or {}
    InputTriggers[self.name] = list
    list[#list + 1] = fn
    return #list
end

function inputProto:removeTrigger(idx)
    local list = InputTriggers[self.name]
    if not list or not list[idx] then return false end
    table.remove(list, idx)
    return true
end

function inputProto:clearTriggers()
    InputTriggers[self.name] = nil
end

function inputProto:enableNetworking(enabled, interval)
    if not SERVER then return false end
    local entry = addNetworkedVariable(self.name, interval)
    if not entry then return false end
    entry.fromInput = true
    entry.enabled = enabled ~= false
    if entry.enabled then
        local v = InputValues[self.name]
        if v ~= nil then updateNetworkedVariable(self.name, v, interval) end
    end
    return entry.enabled
end

local outputProto = {}
outputProto.__index = outputProto

LastOutputs = LastOutputs or {}

function outputProto:update(value)
    if not SERVER then return false end
    if same(value, LastOutputs[self.name]) then return false end
    LastOutputs[self.name] = clone(value)
    wire.ports[self.name] = value
    return true
end

local netVarProto = {}
netVarProto.__index = netVarProto

function netVarProto:get(fallback)
    local v = NetVarValues[self.name]
    if v ~= nil then return v end
    if fallback ~= nil then return fallback end
    return defaultForPortType(self.defaultType or "NORMAL")
end

function netVarProto:addTrigger(fn)
    if type(fn) ~= "function" then return nil end
    local list = NetVarTriggers[self.name] or {}
    NetVarTriggers[self.name] = list
    list[#list + 1] = fn
    return #list
end

function netVarProto:removeTrigger(idx)
    local list = NetVarTriggers[self.name]
    if not list or not list[idx] then return false end
    table.remove(list, idx)
    return true
end

function netVarProto:clearTriggers()
    NetVarTriggers[self.name] = nil
end

function netVarProto:update(value, interval)
    if not SERVER then return false end
    return updateNetworkedVariable(self.name, value, interval)
end

function netVarProto:setInterval(interval)
    if not SERVER then return false end
    return addNetworkedVariable(self.name, interval) ~= nil
end

function netVarProto:changed(keep, slot, epsilon)
    return changed(self:get(), keep, slot or ("__netvar:" .. self.name), epsilon)
end

function netVarProto:resetChanged(slot)
    return resetChanged(slot or ("__netvar:" .. self.name))
end

local function ensureNetVar(name)
    local entry = rawget(NetVars, name)
    if entry then return entry end
    entry = setmetatable({ name = name, interval = 0, dirty = false, enabled = true }, netVarProto)
    rawset(NetVars, name, entry)
    return entry
end

setmetatable(NetVars, {
    __index = function(_, k) if type(k) == "string" then return ensureNetVar(k) end end
})

-- Port parsing
local function parsePortString(str)
    local names, types = {}, {}
    if type(str) ~= "string" then return names, types end
    for token in string.gmatch(str, "[^,%s]+") do
        local name, t = string.match(token, "^([^:]+):?(.*)$")
        if name and name ~= "" then
            names[#names + 1] = name
            types[#types + 1] = normalizePortType(t)
        end
    end
    return names, types
end

local function registerInput(name, portType)
    local entry = rawget(Inputs, name)
    if not entry then
        entry = setmetatable({ name = name }, inputProto)
        rawset(Inputs, name, entry)
    end
    InputTypes[name] = portType
    if InputValues[name] == nil then
        InputValues[name] = defaultForPortType(portType)
    end
    return entry
end

local function registerOutput(name, portType)
    local entry = rawget(Outputs, name)
    if not entry then
        entry = setmetatable({ name = name, portType = portType }, outputProto)
        rawset(Outputs, name, entry)
    end
    return entry
end

function updateInputs(str)
    local names, types = parsePortString(str)
    for i = 1, #names do registerInput(names[i], types[i]) end
    if SERVER then wire.adjustInputs(names, types) end
    return names, types, Inputs
end

function updateOutputs(str)
    local names, types = parsePortString(str)
    for i = 1, #names do registerOutput(names[i], types[i]) end
    if SERVER then wire.adjustOutputs(names, types) end
    return names, types, Outputs
end

function inputClk(name, fn)
    if type(name) ~= "string" or type(fn) ~= "function" then return nil end
    local list = InputTriggers[name] or {}
    InputTriggers[name] = list
    list[#list + 1] = fn
    return #list
end

function netVarClk(name, fn)
    return NetVars[name]:addTrigger(fn)
end

function safeWireCreate(entI, entO, inputName, outputName)
    return pcall(wire.create, entI, entO, inputName, outputName)
end

function safeWireDelete(ent, inputName)
    return pcall(wire.delete, ent, inputName)
end

function updateWirelinkField(wl, field, value, slot)
    if wl == nil or field == nil then return false end
    local key = slot or ("__wl:" .. tostring(wl) .. ":" .. tostring(field))
    if not changed(value, true, key) then return false end
    wl[field] = value
    return true
end

function getNetworkedVariableValue(name, fallback)
    return NetVars[name]:get(fallback)
end

-- Networking targets
function networkTargetSet(target)
    NetworkTargets = target or nil
    return true
end

local function fireTriggers(list, value, name)
    if not list then return end
    for i = 1, #list do
        local fn = list[i]
        xpcall(fn, function(err) print("[SFWN] trigger error: ", err) end, value, name)
    end
end

local function fireNetVarTriggers(name, value)
    fireTriggers(NetVarTriggers[name], value, name)
end

-- Value serialization
local writeValue
writeValue = function(v, depth)
    depth = (depth or 0) + 1
    if depth > 8 then net.writeUInt(TYPE_NIL, 4) return end
    local code = inferTypeCode(v)
    net.writeUInt(code, 4)
    if code == TYPE_NIL then return end
    if code == TYPE_BOOL then net.writeBool(v) return end
    if code == TYPE_NUMBER then net.writeDouble(v) return end
    if code == TYPE_STRING then net.writeString(v) return end
    if code == TYPE_VECTOR then net.writeVector(v) return end
    if code == TYPE_ANGLE then net.writeAngle(v) return end
    if code == TYPE_ENTITY then net.writeEntity(v) return end
    if code == TYPE_ARRAY then
        local n = math.min(#v, 1024)
        net.writeUInt(n, 16)
        for i = 1, n do writeValue(v[i], depth) end
        return
    end
    if code == TYPE_MAP then
        local count = 0
        for _ in pairs(v) do count = count + 1 end
        count = math.min(count, 1024)
        net.writeUInt(count, 16)
        local written = 0
        for k, value in pairs(v) do
            written = written + 1
            if written > count then break end
            writeValue(k, depth)
            writeValue(value, depth)
        end
    end
end

local readValue
readValue = function(depth)
    depth = (depth or 0) + 1
    if depth > 8 then return nil end
    local code = net.readUInt(4)
    if code == TYPE_NIL then return nil end
    if code == TYPE_BOOL then return net.readBool() end
    if code == TYPE_NUMBER then return net.readDouble() end
    if code == TYPE_STRING then return net.readString() end
    if code == TYPE_VECTOR then return net.readVector() end
    if code == TYPE_ANGLE then return net.readAngle() end
    if code == TYPE_ENTITY then return net.readEntity() end
    if code == TYPE_ARRAY then
        local n = net.readUInt(16)
        local out = {}
        for i = 1, n do out[i] = readValue(depth) end
        return out
    end
    if code == TYPE_MAP then
        local n = net.readUInt(16)
        local out = {}
        for i = 1, n do
            local k = readValue(depth)
            local v = readValue(depth)
            if k ~= nil then out[k] = v end
        end
        return out
    end
    return nil
end

local function validRecipient(ply)
    if ply == nil then return true end
    local ok, valid = pcall(isValid, ply)
    return ok and valid == true
end

local function resolveTargets()
    if not ConnectSendEnabled then
        if NetworkTargets == nil then return true, nil end
        if type(NetworkTargets) ~= "table" then
            if validRecipient(NetworkTargets) then return true, NetworkTargets end
            return false, nil
        end
        local out = {}
        for _, p in pairs(NetworkTargets) do
            if validRecipient(p) then out[#out + 1] = p end
        end
        if #out <= 0 then return false, nil end
        return true, out
    end

    local readySet = {}
    for ply in pairs(ConnectSendTargets) do
        if validRecipient(ply) then readySet[ply] = true
        else ConnectSendTargets[ply] = nil end
    end

    if NetworkTargets == nil then
        local out = {}
        for ply in pairs(readySet) do out[#out + 1] = ply end
        if #out <= 0 then return false, nil end
        return true, out
    end

    if type(NetworkTargets) ~= "table" then
        if readySet[NetworkTargets] then return true, NetworkTargets end
        return false, nil
    end

    local out = {}
    for _, p in pairs(NetworkTargets) do
        if readySet[p] then out[#out + 1] = p end
    end
    if #out <= 0 then return false, nil end
    return true, out
end

local function estimateRecordsBits(records)
    local bits = NET_BASE_BITS + estimateStringBits(NET_MESSAGE) + 16
    for i = 1, #records do
        bits = bits + estimateStringBits(records[i][1]) + estimateValueBits(records[i][2])
    end
    return bits
end

local function canSendRecords(records)
    if net == nil or net.getBitsLeft == nil then return true end
    return net.getBitsLeft() >= estimateRecordsBits(records)
end

-- Warn (throttled) when a single send eats more than HALF the remaining net
-- budget -- a heads-up that you are networking too much / too often before it
-- starts dropping or stalling. Just a print; never blocks the send.
local LastBudgetWarn = 0
local function warnHeavySend(records)
    if net == nil or net.getBitsLeft == nil then return end
    local budget = net.getBitsLeft()
    if budget <= 0 then return end
    local estBits = estimateRecordsBits(records)
    if estBits * 2 <= budget then return end   -- under half the budget: fine

    local now = (timer and timer.realtime) and timer.realtime() or 0
    if now - LastBudgetWarn < 5 then return end   -- at most one warning / 5s
    LastBudgetWarn = now
    print(string.format(
        "[SFWN] WARNING: net send is %d bits, over half the %d-bit budget -- reduce networked vars or raise their interval",
        estBits, budget))
end

local function sendRecords(records, targetOverride)
    if #records <= 0 then return true end
    local target
    if targetOverride ~= nil then
        if not validRecipient(targetOverride) then return false, "target" end
        target = targetOverride
    else
        local ok, t = resolveTargets()
        if not ok then return false, "target" end
        target = t
    end

    if not canSendRecords(records) then return false, "budget" end
    warnHeavySend(records)

    local ok, err = xpcall(function()
        net.start(NET_MESSAGE)
        net.writeUInt(#records, 16)
        for i = 1, #records do
            net.writeString(records[i][1])
            writeValue(records[i][2])
        end
        if target == nil then net.send() else net.send(target) end
    end, function(err) return err end)

    if not ok then
        pcall(net.abort)
        print("[SFWN] send error: ", err)
        return false, err
    end

    return true
end

local sendRecordsOrRetry
sendRecordsOrRetry = function(records, targetOverride, attempts)
    local ok, reason = sendRecords(records, targetOverride)
    if ok or reason ~= "budget" or (attempts or 0) <= 0 then return ok, reason end

    timer.simple(FLUSH_INTERVAL, function()
        if targetOverride ~= nil and not validRecipient(targetOverride) then return end
        sendRecordsOrRetry(records, targetOverride, attempts - 1)
    end)

    return false, reason
end

if SERVER then
    local WireSelf = wire.self()
    local DirtyOrder = {}
    local DirtyLookup = {}

    local function markDirty(entry)
        if DirtyLookup[entry.name] then return end
        DirtyLookup[entry.name] = true
        DirtyOrder[#DirtyOrder + 1] = entry
    end

    function addNetworkedVariable(name, interval)
        if type(name) ~= "string" or name == "" then return nil end
        local entry = ensureNetVar(name)
        if interval ~= nil then entry.interval = tonumber(interval) or 0 end
        return entry
    end

    function updateNetworkedVariable(name, value, interval)
        if value == nil then return false end
        local entry = addNetworkedVariable(name, interval)
        if not entry or entry.enabled == false then return false end
        if same(entry.value, value) and not entry.dirty then return false end
        entry.value = clone(value)
        NetVarValues[name] = entry.value
        entry.dirty = true
        markDirty(entry)
        fireNetVarTriggers(name, value)
        return true
    end

    function updateNetworkedVariables(values, interval)
        if type(values) ~= "table" then return 0 end
        local count = 0
        for name, value in pairs(values) do
            if updateNetworkedVariable(name, value, interval) then count = count + 1 end
        end
        return count
    end

    -- Input cache + triggers
    hook.add("input", "SFWN_Input", function(name, value)
        InputValues[name] = value
        InputValidity[name] = WireSelf:isWired(name)
        local entry = rawget(NetVars, name)
        if entry and entry.fromInput and entry.enabled ~= false then
            updateNetworkedVariable(name, value)
        end
        fireTriggers(InputTriggers[name], value, name)
    end)

    -- Flush dirty netvars
    local function flush()
        if #DirtyOrder <= 0 then return end
        local now = CurTime()
        local records, sentEntries, remaining = {}, {}, {}
        for i = 1, #DirtyOrder do
            local entry = DirtyOrder[i]
            DirtyLookup[entry.name] = nil
            if entry.dirty and entry.enabled ~= false then
                local interval = entry.interval or 0
                local due = not entry.lastSent or interval <= 0 or (now - entry.lastSent) >= interval
                if due and #records < MAX_RECORDS_PER_FLUSH then
                    entry.dirty = false
                    entry.lastSent = now
                    records[#records + 1] = { entry.name, entry.value }
                    sentEntries[#sentEntries + 1] = entry
                else
                    remaining[#remaining + 1] = entry
                    DirtyLookup[entry.name] = true
                end
            end
        end
        DirtyOrder = remaining
        if #records <= 0 then return end

        local ok, reason = sendRecords(records)
        if ok or reason ~= "budget" then return end

        for i = 1, #sentEntries do
            local entry = sentEntries[i]
            entry.dirty = true
            entry.lastSent = nil
            if not DirtyLookup[entry.name] then
                DirtyLookup[entry.name] = true
                DirtyOrder[#DirtyOrder + 1] = entry
            end
        end
    end

    timer.create("SFWN_Flush", FLUSH_INTERVAL, 0, flush)

    -- State helpers
    local GUN_FIELDS = {
        { "Name", "acfName", "" },
        { "Ready", "acfReady", false },
        { "AmmoType", "acfAmmoType", "" },
        { "AmmoCount", "acfAmmoCount", 0 },
        { "AmmoCountMax", "acfTotalAmmoCount", 0 },
        { "MagSize", "acfMagSize", 0 },
        { "MagRounds", "acfMagRounds", 0 }
    }
    local GUN_RELOAD_FIELDS = {
        { "ReloadTime", "acfReloadTime", 0 },
        { "ReloadProgress", "acfReloadProgress", 0 },
        { "MaxReloadTime", "acfReloadTimeMax", 0 }
    }
    local PROP_FIELDS = {
        { "Armor", "acfPropArmor", 0 },
        { "ArmorMax", "acfPropArmorMax", 0 }
    }

    local function normalizePrefix(prefix, fallback)
        if type(prefix) == "number" then return fallback .. prefix end
        if type(prefix) == "string" and prefix ~= "" then return prefix end
        return fallback
    end

    local function resolveEntity(value)
        if type(value) == "table" and value.get then value = value:get() end
        if value == nil or type(value) == "number" or type(value) == "string" or type(value) == "boolean" then return nil end
        if isValid(value) then return value end
        return nil
    end

    local function acfValue(ent, method, fallback)
        if not ent then return fallback end
        local fn = ent[method]
        if type(fn) ~= "function" then return fallback end
        local ok, v = pcall(fn, ent)
        if ok and v ~= nil then return v end
        return fallback
    end

    local function readFieldSpec(spec, idx)
        if type(spec) == "string" then return spec, nil, 0, nil end
        if type(spec) == "function" then return "Value" .. idx, nil, 0, spec end
        if type(spec) == "table" then
            return spec.method or spec[1] or spec.name,
                spec.field or spec[2],
                spec.fallback or spec.default or spec[3] or 0,
                spec.fn
        end
    end

    local function fieldSuffix(method, explicit)
        if type(explicit) == "string" and explicit ~= "" then return explicit end
        local s = string.gsub(tostring(method or ""), "^acf", "")
        if s == "" then return tostring(method or "Value") end
        return s
    end

    local function registerStateFields(prefix, fields, interval)
        for i = 1, #fields do
            local f = fields[i]
            local suffix = f[1]
            local fullName = prefix .. suffix
            addNetworkedVariable(fullName, f.interval or interval or 0)
            NetVarValues[fullName] = f[3]
        end
    end

    local function pollGunState(prefix, gun, interval)
        for i = 1, #GUN_FIELDS do
            local f = GUN_FIELDS[i]
            updateNetworkedVariable(prefix .. f[1], acfValue(gun, f[2], f[3]), interval)
        end
        if StateConfigs[prefix] and StateConfigs[prefix].reload then
            for i = 1, #GUN_RELOAD_FIELDS do
                local f = GUN_RELOAD_FIELDS[i]
                updateNetworkedVariable(prefix .. f[1], acfValue(gun, f[2], f[3]), interval)
            end
        end
    end

    local function pollPropState(prefix, prop, interval)
        for i = 1, #PROP_FIELDS do
            local f = PROP_FIELDS[i]
            updateNetworkedVariable(prefix .. f[1], acfValue(prop, f[2], f[3]), interval)
        end
    end

    local function pollCustomState(prefix, ent, interval)
        local cfg = StateConfigs[prefix]
        if not cfg or not cfg.fields then return end
        for i = 1, #cfg.fields do
            local field = cfg.fields[i]
            local value
            if ent then
                if field.fn then
                    local ok, r = pcall(field.fn, ent, prefix, field)
                    if ok and r ~= nil then value = r end
                else
                    value = acfValue(ent, field.method, field.fallback)
                end
            end
            if value == nil then value = field.fallback end
            updateNetworkedVariable(prefix .. field.suffix, value, interval)
        end
    end

    function addNetworkedGunState(prefix, interval, enableReloadTime)
        prefix = normalizePrefix(prefix, "Gun")
        registerStateFields(prefix, GUN_FIELDS, interval)
        if enableReloadTime then registerStateFields(prefix, GUN_RELOAD_FIELDS, interval) end
        StateConfigs[prefix] = { kind = "gun", interval = interval or STATE_POLL_INTERVAL, reload = enableReloadTime == true }
        return true
    end

    function addNetworkedPropArmorState(prefix, interval)
        prefix = normalizePrefix(prefix, "Prop")
        registerStateFields(prefix, PROP_FIELDS, interval)
        StateConfigs[prefix] = { kind = "prop", interval = interval or STATE_POLL_INTERVAL }
        return true
    end

    function addStateSend(prefix, interval, ...)
        prefix = normalizePrefix(prefix, "State")
        local fields = {}
        for i = 1, select("#", ...) do
            local method, fieldName, fallback, fn = readFieldSpec(select(i, ...), i)
            if type(method) == "string" and method ~= "" then
                local suffix = fieldSuffix(method, fieldName)
                fields[#fields + 1] = { suffix = suffix, method = method, fallback = fallback, fn = fn }
                addNetworkedVariable(prefix .. suffix, interval or 0)
                NetVarValues[prefix .. suffix] = fallback
            end
        end
        StateConfigs[prefix] = { kind = "custom", interval = interval or STATE_POLL_INTERVAL, fields = fields }
        return #fields > 0
    end

    function updateNetworkedGunState(prefix, inputOrGun, interval)
        prefix = normalizePrefix(prefix, "Gun")
        if not StateConfigs[prefix] then
            StateConfigs[prefix] = { kind = "gun", interval = interval or STATE_POLL_INTERVAL }
        end
        pollGunState(prefix, resolveEntity(inputOrGun), interval)
        return true
    end

    function updateNetworkedPropArmorState(prefix, inputOrProp, interval)
        prefix = normalizePrefix(prefix, "Prop")
        if not StateConfigs[prefix] then
            StateConfigs[prefix] = { kind = "prop", interval = interval or STATE_POLL_INTERVAL }
        end
        pollPropState(prefix, resolveEntity(inputOrProp), interval)
        return true
    end

    function updateStateSend(prefix, inputOrEntity, interval)
        prefix = normalizePrefix(prefix, "State")
        local cfg = StateConfigs[prefix]
        if not cfg or not cfg.fields then return false end
        pollCustomState(prefix, resolveEntity(inputOrEntity), interval)
        return true
    end

    timer.create("SFWN_StatePoll", STATE_POLL_INTERVAL, 0, function()
        for prefix, cfg in pairs(StateConfigs) do
            local ent = resolveEntity(InputValues[prefix])
            if cfg.kind == "gun" then pollGunState(prefix, ent, cfg.interval)
            elseif cfg.kind == "prop" then pollPropState(prefix, ent, cfg.interval)
            elseif cfg.kind == "custom" then pollCustomState(prefix, ent, cfg.interval) end
        end
    end)

    -- Connect-send
    local function sendSnapshotToPlayer(ply)
        if not validRecipient(ply) then return end
        local records = {}
        for name, value in pairs(NetVarValues) do
            local entry = rawget(NetVars, name)
            if entry == nil or entry.enabled ~= false then
                records[#records + 1] = { name, clone(value) }
                if #records >= MAX_RECORDS_PER_FLUSH then
                    sendRecordsOrRetry(records, ply, SNAPSHOT_RETRY_COUNT)
                    records = {}
                end
            end
        end
        if #records > 0 then sendRecordsOrRetry(records, ply, SNAPSHOT_RETRY_COUNT) end
    end

    net.receive(NET_MESSAGE, function(_, ply)
        pcall(net.readUInt, 16)
        if not ConnectSendEnabled or not validRecipient(ply) then return end
        ConnectSendTargets[ply] = true
        sendSnapshotToPlayer(ply)
    end)

    hook.add("ClientInitialized", "SFWN_ClientInit", function(ply)
        if not ConnectSendEnabled then
            timer.simple(FLUSH_INTERVAL, function()
                if validRecipient(ply) then sendSnapshotToPlayer(ply) end
            end)
        end
    end)

    function enableConnectSend(enabled)
        ConnectSendEnabled = enabled == true
        if not ConnectSendEnabled then ConnectSendTargets = {} end
        return ConnectSendEnabled
    end
end

if CLIENT then
    function addNetworkedVariable(name, interval)
        return ensureNetVar(name)
    end

    function updateNetworkedVariable(name, value, interval)
        if value == nil then return false end
        local entry = ensureNetVar(name)
        local oldValue = NetVarValues[name]
        NetVarValues[name] = clone(value)
        if not same(oldValue, value) then fireNetVarTriggers(name, value) end
        return true
    end

    function updateNetworkedVariables(values)
        if type(values) ~= "table" then return 0 end
        local count = 0
        for name, value in pairs(values) do
            if updateNetworkedVariable(name, value) then count = count + 1 end
        end
        return count
    end

    function addNetworkedGunState() return true end
    function addNetworkedPropArmorState() return true end
    function addStateSend() return true end
    function updateNetworkedGunState() return false end
    function updateNetworkedPropArmorState() return false end
    function updateStateSend() return false end

    function enableConnectSend(enabled)
        ConnectSendEnabled = enabled == true
        if ConnectSendEnabled and not ConnectSendAnnounced then
            timer.create("SFWN_Announce", CONNECT_RETRY_INTERVAL, 0, function()
                if ConnectSendAnnounced then
                    timer.remove("SFWN_Announce")
                    return
                end
                local ok = pcall(function()
                    net.start(NET_MESSAGE)
                    net.writeUInt(0, 16)
                    net.send()
                end)
                if ok then
                    ConnectSendAnnounced = true
                    timer.remove("SFWN_Announce")
                else
                    pcall(net.abort)
                end
            end)
        end
        return ConnectSendEnabled
    end

    local function applyRecord(name, value)
        local entry = ensureNetVar(name)
        local oldValue = NetVarValues[name]
        NetVarValues[name] = clone(value)
        InputValues[name] = NetVarValues[name]
        InputValidity[name] = true
        local didChange = not same(oldValue, value)
        if didChange then
            fireNetVarTriggers(name, value)
            fireTriggers(InputTriggers[name], value, name)
        end
    end

    net.receive(NET_MESSAGE, function()
        local count = net.readUInt(16)
        if count == 0 then return end
        for i = 1, count do
            local name = net.readString()
            local value = readValue()
            applyRecord(name, value)
        end
    end)
end

-- State.<Prefix> reader
local STATE_SUFFIX_MAP = {
    Name = "Name", name = "Name",
    Ready = "Ready", ready = "Ready",
    AmmoType = "AmmoType", ammoType = "AmmoType",
    AmmoCount = "AmmoCount", ammoCount = "AmmoCount",
    AmmoCountMax = "AmmoCountMax", ammoCountMax = "AmmoCountMax",
    MagSize = "MagSize", magSize = "MagSize",
    MagRounds = "MagRounds", magRounds = "MagRounds",
    ReloadTime = "ReloadTime", reloadTime = "ReloadTime",
    ReloadProgress = "ReloadProgress", reloadProgress = "ReloadProgress",
    MaxReloadTime = "MaxReloadTime", maxReloadTime = "MaxReloadTime",
    Armor = "Armor", armor = "Armor",
    ArmorMax = "ArmorMax", armorMax = "ArmorMax",
    PropArmor = "PropArmor", propArmor = "PropArmor",
    PropArmorMax = "PropArmorMax", propArmorMax = "PropArmorMax"
}

local STATE_DEFAULTS = {
    Name = "", Ready = false,
    AmmoType = "", AmmoCount = 0, AmmoCountMax = 0,
    MagSize = 0, MagRounds = 0,
    ReloadTime = 0, ReloadProgress = 0, MaxReloadTime = 0,
    Armor = 0, ArmorMax = 0,
    PropArmor = 0, PropArmorMax = 0
}

local stateProto = {}
stateProto.__index = function(self, key)
    local suffix = STATE_SUFFIX_MAP[key] or key
    local fullName = self.__prefix .. suffix
    local v = NetVarValues[fullName]
    if v ~= nil then return v end
    return STATE_DEFAULTS[suffix]
end

setmetatable(State, {
    __index = function(tbl, key)
        if type(key) ~= "string" then return nil end
        local entry = setmetatable({ __prefix = key }, stateProto)
        rawset(tbl, key, entry)
        return entry
    end
})

return {
    Inputs = Inputs,
    Outputs = Outputs,
    InputValues = InputValues,
    NetVars = NetVars,
    NetVarValues = NetVarValues,
    NetValues = NetValues,
    State = State
}
