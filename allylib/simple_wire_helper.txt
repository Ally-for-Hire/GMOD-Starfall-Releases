--@shared

--[[
Direct-global wire helper layered over starfall_wire_networking_interface.txt.

Wraps the backend so that:
- `updateInputs("Base:ENTITY")` exposes `_G.Base` as a live cached input.
- `updateOutputs("HitPos:VECTOR")` exposes `_G.HitPos`; assigning on the server
  writes the wire output.
- `addNetworkedVariable("Speed", 0.1, 0)` exposes `_G.Speed`; assigning on the
  server queues a replicated update.
- `sendEntityData("Gun", 0.2, "acfReady", "acfState")` polls each method on
  `Inputs.Gun` and replicates the results as `_G.GunacfReady`, `_G.GunacfState`.

Connect-send is enabled by default. Call `enableConnectSend(false)` to broadcast.
]]--

local base = require("starfall_wire_networking_interface.txt")

DirectKinds = DirectKinds or {}
DirectInstalled = DirectInstalled or false
DirectNetworked = DirectNetworked or {}
DirectSendData = DirectSendData or {}

local BaseUpdateInputs = updateInputs
local BaseUpdateOutputs = updateOutputs
local BaseAddNetworkedVariable = addNetworkedVariable
local BaseUpdateNetworkedVariable = updateNetworkedVariable

local function defaultForType(portType)
    if portType == "ENTITY" then return chip() end
    if portType == "VECTOR" then return Vector() end
    if portType == "ANGLE" then return Angle() end
    if portType == "ARRAY" then return {} end
    if portType == "STRING" then return "" end
    if portType == "WIRELINK" then return nil end
    return 0
end

local function entityMethodDefault(method)
    local m = string.lower(tostring(method or ""))
    if string.find(m, "name", 1, true)
        or string.find(m, "type", 1, true)
        or string.find(m, "state", 1, true)
        or string.find(m, "status", 1, true) then return "" end
    if string.find(m, "ready", 1, true)
        or string.find(m, "active", 1, true)
        or string.find(m, "valid", 1, true)
        or string.find(m, "enabled", 1, true) then return false end
    if string.sub(m, 1, 6) == "getpos"
        or string.sub(m, 1, 10) == "getforward"
        or string.sub(m, 1, 8) == "getright"
        or string.sub(m, 1, 5) == "getup"
        or string.sub(m, 1, 11) == "getvelocity" then return Vector() end
    if string.sub(m, 1, 9) == "getangles" then return Angle() end
    return 0
end

local function isUsableEntity(ent)
    if ent == nil then return false end
    local t = type(ent)
    if t == "number" or t == "string" or t == "boolean" then return false end
    local ok, valid = pcall(isValid, ent)
    if ok then return valid == true end
    local fn = ent.isValid
    if type(fn) ~= "function" then return false end
    ok, valid = pcall(fn, ent)
    return ok and valid == true
end

local function installGlobalAccessors()
    if DirectInstalled then return end
    DirectInstalled = true

    local oldMeta = getmetatable(_G) or {}
    local oldIndex = oldMeta.__index
    local oldNewIndex = oldMeta.__newindex

    oldMeta.__index = function(tbl, key)
        local kind = DirectKinds[key]
        if kind == nil then
            if type(oldIndex) == "function" then return oldIndex(tbl, key) end
            if type(oldIndex) == "table" then return oldIndex[key] end
            return nil
        end
        if kind == "input" then return InputValues[key] end
        if kind == "output" then return rawget(LastOutputs, key) end
        return NetVarValues[key]
    end

    oldMeta.__newindex = function(tbl, key, value)
        local kind = DirectKinds[key]
        if kind == nil then
            if type(oldNewIndex) == "function" then oldNewIndex(tbl, key, value)
            elseif type(oldNewIndex) == "table" then oldNewIndex[key] = value
            else rawset(tbl, key, value) end
            return
        end
        if kind == "output" then
            if SERVER and Outputs[key] then Outputs[key]:update(value) end
            return
        end
        if kind == "netvar" then
            if SERVER then BaseUpdateNetworkedVariable(key, value) end
            return
        end
        -- input: ignore assignment
    end

    setmetatable(_G, oldMeta)
end

local function registerDirect(name, kind, defaultValue)
    if type(name) ~= "string" or name == "" then return end
    installGlobalAccessors()
    if DirectKinds[name] == nil then DirectKinds[name] = kind end
    rawset(_G, name, nil)
    if defaultValue == nil then return end
    if DirectKinds[name] == "input" and InputValues[name] == nil then
        InputValues[name] = defaultValue
    elseif DirectKinds[name] == "netvar" and NetVarValues[name] == nil then
        NetVarValues[name] = defaultValue
    end
end

function updateInputs(str)
    local names, types, inputs = BaseUpdateInputs(str)
    for i = 1, #names do
        local name = names[i]
        registerDirect(name, "input", defaultForType(types[i]))
        if SERVER and DirectNetworked[name] == nil then
            local input = Inputs[name]
            if input then
                input:enableNetworking(true)
                DirectNetworked[name] = true
            end
        end
    end
    return names, types, inputs
end

function updateOutputs(str)
    local names, types, outputs = BaseUpdateOutputs(str)
    for i = 1, #names do
        registerDirect(names[i], "output", defaultForType(types[i]))
    end
    return names, types, outputs
end

function updateOutput(name, value)
    if type(name) ~= "string" or not Outputs[name] then return false end
    return Outputs[name]:update(value)
end

function updateOutputValues(values)
    if type(values) ~= "table" then return 0 end
    local count = 0
    for name, value in pairs(values) do
        if updateOutput(name, value) then count = count + 1 end
    end
    return count
end

local function parseNames(value)
    local names = {}
    if value == nil or value == true then
        for name, kind in pairs(DirectKinds) do
            if kind == "input" then names[#names + 1] = name end
        end
        return names
    end
    if type(value) == "string" then
        for token in string.gmatch(value, "[^,%s]+") do
            local name = string.match(token, "^([^:]+)")
            if name and name ~= "" then names[#names + 1] = name end
        end
        return names
    end
    if type(value) == "table" then
        for k, v in pairs(value) do
            if type(v) == "string" and v ~= "" then names[#names + 1] = v
            elseif type(k) == "string" and k ~= "" then names[#names + 1] = k end
        end
    end
    return names
end

function networkInputs(namesOrAll, interval)
    if not SERVER then return 0 end
    local names = parseNames(namesOrAll)
    local count = 0
    for i = 1, #names do
        local input = Inputs[names[i]]
        if input then
            input:enableNetworking(true, interval)
            DirectNetworked[names[i]] = true
            count = count + 1
        end
    end
    return count
end

function networkInput(name, interval)
    return networkInputs({ name }, interval) == 1
end

function unnetworkInputs(namesOrAll)
    if not SERVER then return 0 end
    local names = parseNames(namesOrAll)
    local count = 0
    for i = 1, #names do
        local input = Inputs[names[i]]
        if input then
            input:enableNetworking(false)
            DirectNetworked[names[i]] = false
            count = count + 1
        end
    end
    return count
end

function addNetworkedVariable(name, interval, defaultValue)
    local entry = BaseAddNetworkedVariable(name, interval)
    if not entry then return nil end
    if defaultValue == nil then defaultValue = NetVarValues[name] end
    if defaultValue == nil then defaultValue = 0 end
    local existingKind = DirectKinds[name]
    local hadValue = NetVarValues[name] ~= nil
    registerDirect(name, "netvar", defaultValue)
    if SERVER and existingKind ~= "input" and not hadValue then
        BaseUpdateNetworkedVariable(name, defaultValue, interval)
    end
    return entry
end

function networkVariable(name, interval, defaultValue)
    return addNetworkedVariable(name, interval, defaultValue)
end

function networkVariables(valuesOrNames, interval)
    if type(valuesOrNames) == "table" then
        local count = 0
        for name, value in pairs(valuesOrNames) do
            if type(name) == "string" then
                addNetworkedVariable(name, interval, value)
                count = count + 1
            elseif type(value) == "string" then
                addNetworkedVariable(value, interval)
                count = count + 1
            end
        end
        return count
    end
    local names = parseNames(valuesOrNames)
    local count = 0
    for i = 1, #names do
        addNetworkedVariable(names[i], interval)
        count = count + 1
    end
    return count
end

function directValue(name, defaultValue)
    if type(name) ~= "string" or name == "" then return false end
    registerDirect(name, "value", defaultValue)
    return true
end

function getDirectValue(name, fallback)
    local v = NetVarValues[name]
    if v ~= nil then return v end
    v = InputValues[name]
    if v ~= nil then return v end
    return fallback
end

function changedClk(name, fn)
    if type(name) ~= "string" or type(fn) ~= "function" then return nil end
    local kind = DirectKinds[name]
    if kind == "input" then return inputClk(name, fn) end
    return netVarClk(name, fn)
end

function sendEntityData(name, interval, ...)
    if type(name) ~= "string" or name == "" then return false end
    local methods = {}
    for i = 1, select("#", ...) do
        local m = select(i, ...)
        if type(m) == "string" and m ~= "" then methods[#methods + 1] = m end
    end
    if #methods == 0 then return false end
    interval = tonumber(interval) or 0

    for i = 1, #methods do
        addNetworkedVariable(name .. methods[i], interval, entityMethodDefault(methods[i]))
    end

    DirectSendData[name] = { methods = methods, interval = interval }

    if SERVER then
        local timerName = "SWH_sendData_" .. name
        local pollInterval = interval > 0 and interval or 0.1
        timer.create(timerName, pollInterval, 0, function()
            local ent = InputValues[name]
            local valid = isUsableEntity(ent)
            for i = 1, #methods do
                local method = methods[i]
                if valid then
                    local fn = ent[method]
                    local ok, v = false, nil
                    if type(fn) == "function" then ok, v = pcall(fn, ent) end
                    if ok and v ~= nil then
                        BaseUpdateNetworkedVariable(name .. method, v, interval)
                    end
                end
            end
        end)
    end
    return true
end

enableConnectSend(true)

return base
