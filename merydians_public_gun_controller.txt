require("allylib/general_algorithms.txt")
require("allylib/simple_wire_helper.txt")
require("allylib/merys_integrated_ballistics_computer.txt")

updateInputs("Lock:NORMAL Active:NORMAL Gun:ENTITY Base:ENTITY HitPos:VECTOR LoadPitch:NORMAL")
updateOutputs("TurretActive:NORMAL HAimPos:VECTOR VAimPos:VECTOR TTA:NORMAL")

CurTime = timer.curtime
ZERO_VEC = Vector()

ACTIVE_INTERVAL = 0.05
IDLE_INTERVAL = 0.5
LOCK_CLICK_DEBOUNCE = 0.15
WIRING_CHECK_INTERVAL = 0.5

LastGun = nil
Locked = false
LastLock = false
LastLockClick = -1
NextWiringCheck = 0
LoadPitchWired = false

local function liveInput(name, fallback)
    local input = Inputs[name]
    local value = input and input:get() or InputValues[name]
    if value ~= nil then return value end
    return fallback
end

hook.add("input", "SF_MPGC_Input", function(name, value)
    if name == "Lock" then
        -- Rising-edge AND debounce (both required), matching the private
        -- controller. The old `LastLock == 0 or ...` bypassed the debounce
        -- because the input hook only fires on change, so LastLock was always 0
        -- on a fresh press -> every click toggled regardless of LOCK_CLICK_DEBOUNCE.
        local lock = (value or 0) ~= 0
        local now = CurTime()
        if lock and not LastLock and now - LastLockClick >= LOCK_CLICK_DEBOUNCE then
            Locked = not Locked
            LastLockClick = now
        end
        LastLock = lock
    end
    if name == "Gun" then LastGun = nil end
end)

-- Load-pitch fallback aim: when the ballistic solver has no static solution,
-- point the gun along the base pitch minus the load pitch. Was duplicated in
-- both solve branches below; pulled out here (mirrors the private controller's
-- CFBA_ReloadAim helper).
local function loadPitchAim(aimPos, gun, base, loadPitch)
    if loadPitch == nil then return aimPos end
    local origin = getGunOrigin(gun)
    if not origin then return aimPos end
    local yaw = (aimPos - origin):getAngle().y
    return origin + Angle(base:getAngles().p - loadPitch, yaw, 0):getForward() * 10000
end

local function main()
    local now = CurTime()
    if now >= NextWiringCheck then
        LoadPitchWired = wire.self():isWired("LoadPitch")
        NextWiringCheck = now + WIRING_CHECK_INTERVAL
    end

    Outputs.TurretActive:update(not Locked and 1 or 0)

    local gun = liveInput("Gun")
    local base = liveInput("Base")
    local aimPos = liveInput("HitPos", ZERO_VEC)
    local loadPitch = LoadPitchWired and liveInput("LoadPitch") or nil
    local tta = 0

    if gun and gun:isValid() and gun ~= LastGun then
        if CFBA_SetGun then CFBA_SetGun(gun) end
        LastGun = gun
    end

    if gun and gun:isValid() and base and base:isValid() and aimPos:getLength() > 0 then
        local solved = false
        if CFBA_SolveStatic then
            local _, staticAimPos, staticTta = CFBA_SolveStatic(base, aimPos)
            if hasVectorValue(staticAimPos) then
                aimPos = staticAimPos
                tta = staticTta
                solved = true
            end
        end
        if not solved then
            aimPos = loadPitchAim(aimPos, gun, base, loadPitch)
        end
    end

    Outputs.HAimPos:update(aimPos)
    Outputs.VAimPos:update(aimPos)
    Outputs.TTA:update(tta)
end

timer.create("SF_MPGC", ACTIVE_INTERVAL, 0, main)
